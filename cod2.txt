import threading
import time

start = time.perf_counter()


def do_something(seconds): #attributing an argument to the function
	print(f'Sleeping {seconds} second(s)...')
	time.sleep(seconds)
	print('Done Sleeping')

threads = [] #creating a list of threads

for _ in range(10): # underscore means it's a throwaway variable
	t = threading.Thread(target=do_something, args= [1.5]) #passing the arguments as a list
	t.start()
	threads.append(t)

for thread in threads:
	thread.join()


finish = time.perf_counter()

print(f' Finished in {round(finish-start, 2)} second(s)')
